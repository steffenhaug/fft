\usetikzlibrary{decorations.pathreplacing,angles,quotes}

\def\GRIDSIZE{1.625}
\draw[<->] (0, -\GRIDSIZE) -- (0, \GRIDSIZE)
    node[left] {\tiny y};
\draw[<->] (-4.0*\GRIDSIZE, 0) -- (4.0*\GRIDSIZE, 0)
    node[below] {\tiny x};

\def\TICKWIDTH{.05}

\foreach \n in {-3, -1, 1, 3} {
    \draw (2*\n, \TICKWIDTH) -- (2*\n, -\TICKWIDTH);
    \draw (-2*\n, \TICKWIDTH) -- (-2*\n, -\TICKWIDTH);
}

\only<1-2>{
    \draw [\MyOrange,thick] plot [smooth,tension=1]
        coordinates { (0,0.1) (0.5, 0.8) (1,0.2) (1.5, 0.4) (2,0.5) };
}

\only<2>{
\draw [\MyOrange,thick,rotate=180] plot [smooth,tension=1]
    coordinates { (0,0.1) (0.5, 0.8) (1,0.2) (1.5, 0.4) (2,0.5) };

    \foreach \n in {-1, 1} {

        \draw [gray,shift={(4*\n, 0)}] plot [smooth,tension=1]
            coordinates { (0,0.1) (0.5, 0.8) (1,0.2) (1.5, 0.4) (2,0.5) };
        \draw [gray,rotate=180,shift={(4*\n,0)}] plot [smooth,tension=1]
            coordinates { (0,0.1) (0.5, 0.8) (1,0.2) (1.5, 0.4) (2,0.5) };
    }
}

\only<3> {
    \foreach \n in {-1, 0, 1} {
        \draw [SuperLightGray,shift={(4*\n, 0)}] plot [smooth,tension=1]
            coordinates { (0,0.1) (0.5, 0.8) (1,0.2) (1.5, 0.4) (2,0.5) };
        \draw [SuperLightGray,rotate=180,shift={(4*\n,0)}] plot [smooth,tension=1]
            coordinates { (0,0.1) (0.5, 0.8) (1,0.2) (1.5, 0.4) (2,0.5) };

        \draw[\MyOrange, domain=-2:2, thick, smooth, variable=\x]
            plot ({\x}, {sin(deg(pi*\x/2))});

        \draw[fill=white] (0,0) circle (.05);
        \draw[fill=white] (2,0) circle (.05);
    }
}

\only<4> {
    \foreach \n in {-1, 0, 1} {
        \draw [SuperLightGray,shift={(4*\n, 0)}] plot [smooth,tension=1]
            coordinates { (0,0.1) (0.5, 0.8) (1,0.2) (1.5, 0.4) (2,0.5) };
        \draw [SuperLightGray,rotate=180,shift={(4*\n,0)}] plot [smooth,tension=1]
            coordinates { (0,0.1) (0.5, 0.8) (1,0.2) (1.5, 0.4) (2,0.5) };


        \draw[\MyOrange, domain=-2:2, thick, smooth, variable=\x]
            plot ({\x}, {sin(deg(pi*\x))});

        \draw[fill=white] (0,0) circle (.05);
        \draw[fill=white] (2,0) circle (.05);
    }
}

\only<5-> {
    \foreach \n in {-1, 0, 1} {
        \draw [SuperLightGray,shift={(4*\n, 0)}] plot [smooth,tension=1]
            coordinates { (0,0.1) (0.5, 0.8) (1,0.2) (1.5, 0.4) (2,0.5) };
        \draw [SuperLightGray,rotate=180,shift={(4*\n,0)}] plot [smooth,tension=1]
            coordinates { (0,0.1) (0.5, 0.8) (1,0.2) (1.5, 0.4) (2,0.5) };


        \draw[\MyOrange, samples=200,domain=-2:2, thick, smooth, variable=\x]

        \draw[fill=white] (0,0) circle (.05);
        \draw[fill=white] (2,0) circle (.05);
    }
